var ch as string = ""

var comp as integer = 0

label cont

n = sr::Peek()

if n < 0 then
goto cont
end if

place loop

chr = $char$n
ch = $string$chr
comp = String::Compare(ch, Utils.Constants::cr)

if comp = 0 then
crflag = true
end if

comp = String::Compare(ch, Utils.Constants::lf)

if comp = 0 then
lfflag = true
end if

andflg = crflag and lfflag
orflg = crflag or lfflag

if orflg = false then
buf = String::Concat(buf, ch)
else
if lfflag = true then
curline++
//Console::WriteLine(buf)
curstmt = new Stmt()
curstmt::Line = curline
curln = new Line()

curstmtlen = curstmt::Tokens[l]

if curstmtlen <> 0 then
stmts::AddStmt(curstmt)
end if

buf = ""
crflag = false
lfflag = false
end if
end if

n = sr::Peek()

if n = -1 then
curline++
//Console::WriteLine(buf)
curstmt = new Stmt()
curstmt::Line = curline
curln = new Line()
curstmt = curln::Analyze(curstmt, buf)
curstmtlen = curstmt::Tokens[l]

if curstmtlen <> 0 then
stmts::AddStmt(curstmt)
end if

goto cont
else
goto loop
end if

place cont

sr::Close()

return stmts
end method

end class
